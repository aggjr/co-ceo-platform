Modelagem Arquitetural do Banco de Dados co-CEO Core

💡 Nova Adição: Telemetria e Monitoramento de Uso

Conforme solicitado, incluímos a tabela telemetry_events. É fundamental entender que:

  • audit_logs: Foco em Compliance. Grava apenas ações que mutaram os dados (INSERT, UPDATE, DELETE) com um DIFF de JSON.
  • telemetry_events: Foco em Produto e UX. Grava visualizações de página, cliques, tempo de sessão e navegação entre módulos. Isso gera os KPIs de engajamento sem sujar as tabelas de negócio e compliance. (Nota: no futuro, em altíssima escala, essa tabela geralmente é migrada para um banco NoSQL temporal como Elasticsearch ou ClickHouse).

Script SQL Consolidado

-- ====================================================================================
-- co-CEO Core Database Schema (v1.1 - N-Level Hierarchy + Telemetry)
-- ====================================================================================

-- 1. ESTRUTURA HIERÁRQUICA INFINITA (MATERIALIZED PATH)
CREATE TABLE organizations (
    id VARCHAR(36) PRIMARY KEY,
    parent_id VARCHAR(36) NULL,           -- NULL = Raiz Máxima
    name VARCHAR(255) NOT NULL,
    type VARCHAR(50) NOT NULL,            -- 'brand', 'region', 'franchisee', 'store'
    document VARCHAR(50),
    path TEXT NOT NULL,                   -- Pulo do Gato: "/root_id/node1_id/node2_id/"
    status ENUM('active', 'suspended', 'closed') DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (parent_id) REFERENCES organizations(id) ON DELETE RESTRICT
);

-- 2. IDENTIDADE E USUÁRIOS
CREATE TABLE users (
    id VARCHAR(36) PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    full_name VARCHAR(255) NOT NULL,
    preferred_name VARCHAR(100),
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 3. RBAC (ROLE-BASED ACCESS CONTROL)
CREATE TABLE roles (
    id VARCHAR(36) PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    scope ENUM('global', 'node') NOT NULL, 
    description VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE permissions (
    id VARCHAR(36) PRIMARY KEY,
    module VARCHAR(50) NOT NULL,          -- 'core', 'cash', 'stockspin'
    code VARCHAR(100) UNIQUE NOT NULL     -- 'cash:incomes:write'
);

CREATE TABLE role_permissions (
    role_id VARCHAR(36) NOT NULL,
    permission_id VARCHAR(36) NOT NULL,
    PRIMARY KEY (role_id, permission_id),
    FOREIGN KEY (role_id) REFERENCES roles(id) ON DELETE CASCADE,
    FOREIGN KEY (permission_id) REFERENCES permissions(id) ON DELETE CASCADE
);

-- 4. O VÍNCULO MÁGICO (USUÁRIO + ROLE + CONTEXTO)
CREATE TABLE user_roles (
    id VARCHAR(36) PRIMARY KEY,
    user_id VARCHAR(36) NOT NULL,
    role_id VARCHAR(36) NOT NULL,
    organization_id VARCHAR(36) NULL,     -- Se NULL e role for global, vê TUDO.
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (role_id) REFERENCES roles(id) ON DELETE RESTRICT,
    FOREIGN KEY (organization_id) REFERENCES organizations(id) ON DELETE CASCADE
);

-- 5. TRILHA DE AUDITORIA ESTRITA (LOG DE MUTAÇÃO DE DADOS)
CREATE TABLE audit_logs (
    id BIGINT AUTO_INCREMENT PRIMARY KEY, 
    table_name VARCHAR(100) NOT NULL,
    record_id VARCHAR(36) NOT NULL,
    action ENUM('INSERT', 'UPDATE', 'DELETE', 'SOFT_DELETE') NOT NULL,
    organization_id VARCHAR(36) NULL,
    actor_user_id VARCHAR(36) NOT NULL,
    impersonator_user_id VARCHAR(36) NULL,
    old_payload JSON NULL,
    new_payload JSON NULL,
    ip_address VARCHAR(45),
    user_agent TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_audit_record (table_name, record_id),
    INDEX idx_audit_org (organization_id),
    INDEX idx_audit_actor (actor_user_id)
);

-- 6. NOVO: TELEMETRIA E USO DA PLATAFORMA (PRODUTO)
CREATE TABLE telemetry_events (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    organization_id VARCHAR(36) NULL,     -- Onde o usuário estava navegando
    user_id VARCHAR(36) NOT NULL,         -- Quem clicou/visualizou
    session_id VARCHAR(100),              -- Identificador único do login atual
    
    event_type VARCHAR(50) NOT NULL,      -- 'page_view', 'button_click', 'module_open'
    module VARCHAR(50) NOT NULL,          -- 'cash', 'stockspin', 'core_admin'
    screen VARCHAR(100) NOT NULL,         -- 'dashboard', 'settings', 'reports'
    action_detail VARCHAR(255),           -- 'Clicou em exportar PDF', 'Abriu modal'
    
    metadata JSON NULL,                   -- { "duration_ms": 1200, "browser": "Chrome" }
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    -- Índices cruciais para relatórios analíticos de uso rápidos
    INDEX idx_telem_org (organization_id),
    INDEX idx_telem_user (user_id),
    INDEX idx_telem_module_screen (module, screen),
    INDEX idx_telem_date (created_at)
);
